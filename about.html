סִפּוּר הַמּוּתָג

הַחֹמֶר הַנָּכוֹן
לַפְּרוֹיֶקְט הַנָּכוֹן.

מִקוּם שֶׁמַּטָּרָתוֹ לִבְחֹר אֶת הָאֶבֶן וְהַשַּׁיִשׁ הָאֵיכוּתִיִּים בְּיוֹתֵר, וּלְחַבֵּר אוֹתָם לַדִּיּוּק הָאַדְרִיכָלִי שֶׁכָּל פְּרוֹיֶקְט מֵשִׂים.

"אֶבֶן. חוֹמֶר שֶׁלֹּא נוֹעַד לְהִמָּכֵר. הוּא נוֹעַד לְהַחֲזִיק."

הַסִּפּוּר

סְטוּדְיוֹ שֵׁש שֵׁש הוּקַם מִתּוֹךְ אַהֲבָה עֲמֻקָּה לַחוֹמֶר הַגּוֹלְמִי. לֹא לְמוּצָר. לַחוֹמֶר עַצְמוֹ — לַגִּידִים הַצִּבְעוֹנִיִּים שֶׁל הַשַּׁיִשׁ, לַמֶּרְקָם הָאוֹרְגָנִי שֶׁל הָאֶבֶן, לְסִפּוּר הַגֵּיאוֹלוֹגִי שֶׁכָּל לוּחַ נוֹשֵׂא בְּתוֹכוֹ.

אֲנַחְנוּ מַאֲמִינִים שֶׁאֶבֶן טוֹבָה אֵינָהּ רַק חֹמֶר בִּנְיָּן — הִיא גִּבּוּשׁ שֶׁל אַלְפֵי שָׁנוֹת לַחַץ גֵּיאוֹלוֹגִי, טֶמְפֶּרָטוּרוֹת קִיצוֹנִיּוֹת, וְעַבּוֹדָה שֶׁל הַיְּצִירָה. כְּשֶׁמְּמַקְמִים אוֹתָהּ נְכוֹן בְּחָלָל — הִיא מְדַבֶּרֶת.

הַסְּטוּדְיוֹ מְשַׁמֵּשׁ כְּגֶשֶׁר בֵּין הַחוֹמֶר הַגּוֹלְמִי לְבֵין הַחֲזוֹן הָאַדְרִיכָלִי. כָּל פְּגִישָׁה עִם לָקוֹחַ מַתְחִילָה בְּשְׁאֵלָה אַחַת: מַה הַפְּרוֹיֶקְט מְבַקֵּשׁ לֵהָגֵּד?

הָעֶרְכִּים שֶׁלָּנוּ

אֲמִינוּת הַחוֹמֶר

אֲנַחְנוּ עוֹבְדִים אַךְ וְרַק עִם חוֹמָרִים טִבְעִיִּים אֲמִיתִּיִּים. אֵין חִיקּוּיִים, אֵין תְּחֲלִיפִים. הָאֶבֶן שֶׁמַּגִּיעָה לְאַתַּר הַבִּנְיָּן הִיא בְּדִיּוּק מַה שֶּׁנִּרְאֶה בְּסְטוּדְיוֹ.

שִׁיתּוּף פְּעֻלָּה אַדְרִיכָלִי

אֲנַחְנוּ לֹא מוֹכְרִים חוֹמָרִים — אֲנַחְנוּ מְשַׁתְּפִים פְּעֻלָּה עִם אַדְרִיכָלִים וּמְעַצְּבִים לְאֹרֶךְ כָּל הַתַּהֲלִיךְ, מֵהַשִּׂרְטוּט עַד לְהַרְכָּבָה.

דִּיּוּק מִלִּימֶטְרִי

כָּל חִיתּוּךְ, כָּל קֶצֶה, כָּל חֲרִיטָה — מְבֻצַּעַת לְפִי תָּכְנִיּוֹת מְדֻיָּקוֹת. הַפֶּרֶט הַקָּטָן בְּיוֹתֵר הוּא חֵלֶק בִּלְתִּי נִפְרָד מֵהַתּוֹצָאָה הַסּוֹפִית.

נִיסָיוֹן שֶׁנִּבְנָה בַּשָּׂטַח

עֶשְׂרִים שָׁנָה שֶׁל פְּרוֹיֶקְטִים מְגַבִּים כָּל הַמְלָצָה. כָּל חוֹמֶר שֶׁאֲנַחְנוּ מַצִּיעִים נִבְדַּק, הוּכַּח וְיָדַעְנוּ אֵיךְ הוּא מִתְנַהֵג לְאֹרֶךְ זְמַן.

הַמָּקוֹם

הַסְּטוּדְיוֹ שֶׁלָּנוּ שׁוֹכֵן בְּכוֹכַב מִיכָאֵל — אֵזוֹר יִשּׁוּבִי בַּדָּרוֹם, הַמְּשַׁמֵּשׁ בָּסִיס נוֹחַ לִפְרוֹיֶקְטִים בְּכָל רְחֲבֵי הָאֶזוֹר.

הַמִּפְעָל מְצֻיָּד בְּטֶכְנוֹלוֹגִיָּה מֻדֶּרֶנֶת לְחִיתּוּךְ וְעִיבּוּד אֶבֶן, לְצַד צֶמֶד יָדַיִם מְנֻסּוֹת שֶׁמַּכִּירוֹת אֶת הַחוֹמֶר מִן הַשּׁוּרֶשׁ.
